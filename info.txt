# Pobranie CUnit

wget -nd https://downloads.sourceforge.net/project/cunit/CUnit/2.1-3/CUnit-2.1-3.tar.bz2

# Rozpakowanie, kompilacja i instalacja
tar xvf CUnit-2.1-3.tar.bz2
cd CUnit-2.1-3
autoreconf -i
./configure --prefix=/home/student/lab3/cunit
make -j5
make install
